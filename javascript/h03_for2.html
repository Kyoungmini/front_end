2차원 배열 문제

const nums=[10,2,1,-4,29,50,0];
const sort_nums=[]

sort_nums=[50,29,10,2,1,0,-4] 의 결과를 만드세요 (선택정렬)

  1. max=50; nums=[10,2,1,-4,29,0]; sort_nums=[50];
  2. max=29; nums=[10,2,1,-4,0]; sort_nums=[50,29];
  3. max=10; nums=[2,1,-4,0]; sort_nums=[50,29,10];
  4. max=2; nums=[1,-4,0]; sort_nums=[50,29,10,2];
  5. max=1; nums=[-4,0]; sort_nums=[50,29,10,2,1];
  6. max=0; nums=[-4]; sort_nums=[50,29,10,2,1,0];
  7. max=-4; nums=[]; sort_nums=[50,29,10,2,1,0,-4];

규칙을 찾아서 2차원 배열식으로 바꾸세요(반복문을 2중으로 만드세요.)

const nums2=[10,2,1,-4,29,50,0]; => nums2=[50,29,10,2,1,0,-4];

  1. max=50; temp=num2[0] num2=[50, 2,1,-4,29,10,0]
  2. max=29; temp=num2[1] num2=[50,29, 1,-4,2,10,0]
  3. max=10; temp=num2[2] num2=[50,29,10, -4,2,1,0]
  4. max=2; temp=num2[3] num2=[50,29,10,2, -4,1,0]
  5. max=1; temp=num2[4] num2=[50,29,10,2,1, -4,0]
  6. max=0; temp=num2[5] num2=[50,29,10,2,1,0, -4]
  7. max=-4; temp=num2[6] num2=[50,29,10,2,1,0,-4 ]

const nums2=[-4,2,1,10]; => nums2=[10,2,1,-4]; (버블정렬)

앞뒤 값을 비교해서 더 큰것을 옮긴다.(밀어내기 정렬)

  1. num2[0]>nums[1] [2,-4,1,10]
  2. num2[1]>nums[2] [2,1,-4,10]
  3. num2[2]>nums[3] [2,1,10,-4]
  4. num2[0]>nums[1] [2,1,10,-4]
  5. num2[0]>nums[1] [2,10,1,-4]
  6. num2[0]>nums[1] [10,2,1,-4]
